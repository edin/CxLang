CX parser grammar draft
=======================

This file is a human-readable source of truth for the hand-written parser.
It is intentionally BNF-like rather than generator-specific. Keep it close to
the parser while the compiler is evolving.

Conventions
-----------

    rule            = definition ;
    a?              = optional a
    a*              = zero or more
    a+              = one or more
    a | b           = alternatives
    TOKEN           = lexer token
    "text"          = literal token text


Program
-------

    program
        = module_decl? import_decl* top_level_decl* EOF ;

    module_decl
        = "module" module_path ";" ;

    module_path
        = IDENTIFIER ("." IDENTIFIER)* ;

    import_decl
        = "import" module_path import_alias? ";"
        | "from" module_path "import" symbol_import_list ";" ;

    import_alias
        = "as" IDENTIFIER ;

    symbol_import_list
        = symbol_import ("," symbol_import)* ;

    symbol_import
        = IDENTIFIER import_alias? ;

    include_decl
        = "include" "<" include_path ">" ";" ;


Top-Level Declarations
----------------------

    top_level_decl
        = attributes? module_decl
        | attributes? import_decl
        | attributes? include_decl
        | attributes? declare_block
        | attributes? extern_function_decl
        | attributes? type_decl
        | attributes? requirement_decl
        | attributes? enum_decl
        | attributes? interface_decl
        | attributes? struct_decl
        | attributes? extension_decl
        | attributes? union_decl
        | attributes? global_decl
        | attributes? function_decl
        | attributes? test_decl ;

    attributes
        = attribute_application+ ;

    attribute_application
        = "@" IDENTIFIER ("(" attribute_argument_list? ")")? ;

    attribute_argument_list
        = attribute_argument ("," attribute_argument)* ;

    attribute_argument
        = IDENTIFIER ":" attribute_value
        | attribute_value ;

    attribute_value
        = tokens-until-top-level-comma-or-rparen ;


C Declarations
--------------

    declare_block
        = "declare" "<" include_path ">" "{" c_declare_member* "}" ;

    c_declare_member
        = c_link_decl
        | type_decl
        | const_decl
        | macro_const_decl
        | macro_function_decl
        | struct_decl
        | enum_decl
        | raw_union_decl
        | extern_function_decl ;

    c_link_decl
        = "link" IDENTIFIER? STRING ";" ;

    extern_function_decl
        = "extern"? "fn" IDENTIFIER "(" parameter_list? ")" "->" type variadic_marker? ";" ;

    variadic_marker
        = "," "..." ;

    macro_const_decl
        = "macro" "const" IDENTIFIER ":" type ";" ;

    macro_function_decl
        = "macro" "fn" IDENTIFIER "(" parameter_list? ")" "->" type ";" ;


Types
-----

    type
        = function_type
        | named_type type_suffix* ;

    named_type
        = "const"? qualified_type_name generic_arguments?
        | c_tag_type ;

    c_tag_type
        = ("struct" | "enum" | "union") IDENTIFIER ;

    qualified_type_name
        = IDENTIFIER ("." IDENTIFIER)* type_name_continuation* ;

    type_name_continuation
        = "void" | "char" | "short" | "int" | "long" | "float" | "double" ;

    generic_arguments
        = "<" type ("," type)* ">" ;

    type_suffix
        = "*"
        | "[" expression-text? "]" ;

    function_type
        = "fn" "(" function_type_parameter_list? ")" "->" type ;

    function_type_parameter_list
        = function_type_parameter ("," function_type_parameter)* ;

    function_type_parameter
        = IDENTIFIER ":" type
        | type
        | "..." ;


Type, Requirement, Interface, Struct
------------------------------------

    type_decl
        = "type" IDENTIFIER generic_parameter_list? "=" type ";"
        | "type" IDENTIFIER generic_parameter_list? "using" type adapter_body? ;

    adapter_body
        = "{" adapter_member* "}" ;

    adapter_member
        = "expose" IDENTIFIER expose_alias? expose_return_type? ";"
        | function_decl ;

    expose_alias
        = "as" IDENTIFIER ;

    expose_return_type
        = "->" type ;

    generic_parameter_list
        = "<" generic_parameter ("," generic_parameter)* ">" ;

    generic_parameter
        = IDENTIFIER requirement_bound_list? ;

    requirement_bound_list
        = ":" requirement_bound ("+" requirement_bound)* ;

    requirement_bound
        = qualified_type_name generic_arguments? ;

    requirement_decl
        = "requires" IDENTIFIER generic_parameter_list? "{" requirement_member* "}" ;

    requirement_member
        = requirement_field
        | requirement_function ;

    requirement_field
        = IDENTIFIER ":" type ";" ;

    requirement_function
        = "static"? "fn" IDENTIFIER "(" parameter_list? ")" "->" type ";" ;

    interface_decl
        = "interface" IDENTIFIER generic_parameter_list? "{" interface_method* "}" ;

    interface_method
        = "fn" IDENTIFIER "(" parameter_list? ")" "->" type ";" ;

    struct_decl
        = "struct" IDENTIFIER generic_parameter_list? requirement_impl_list? "{" struct_member* "}" ;

    requirement_impl_list
        = ":" requirement_bound ("," requirement_bound)* ;

    struct_member
        = struct_field
        | function_decl
        | static_function_decl ;

    struct_field
        = IDENTIFIER ":" type ";" ;

    extension_decl
        = "extension" type where_clause? "{" extension_member* "}" ;

    extension_member
        = function_decl
        | static_function_decl ;

    where_clause
        = "where" requirement_bound ("," requirement_bound)* ;


Enums and Unions
----------------

    enum_decl
        = "enum" IDENTIFIER "{" enum_member_list? "}" ;

    enum_member_list
        = enum_member ("," enum_member)* ","? ;

    enum_member
        = IDENTIFIER ("=" expression)? ;

    union_decl
        = "union" IDENTIFIER generic_parameter_list? "{" union_variant* "}" ;

    raw_union_decl
        = "raw" "union" IDENTIFIER "{" struct_field* "}" ;

    union_variant
        = IDENTIFIER ":" type ";" ;


Functions and Globals
---------------------

    function_decl
        = "fn" function_name generic_parameter_list? "(" parameter_list? ")" return_type? function_body ;

    static_function_decl
        = "static" function_decl ;

    function_name
        = IDENTIFIER
        | type "." IDENTIFIER ;

    return_type
        = "->" type ;

    function_body
        = block
        | "=>" expression ";" ;

    parameter_list
        = parameter ("," parameter)* ;

    parameter
        = IDENTIFIER ":" type ;

    global_decl
        = ("let" | "const") IDENTIFIER optional_type initializer? ";" ;

    const_decl
        = "const" IDENTIFIER ":" type initializer? ";" ;

    optional_type
        = (":" type)? ;

    initializer
        = "=" expression ;

    test_decl
        = "test" STRING block ;


Statements
----------

    block
        = "{" statement* "}" ;

    statement
        = block
        | let_stmt
        | const_stmt
        | return_stmt
        | if_stmt
        | while_stmt
        | for_stmt
        | foreach_stmt
        | switch_stmt
        | match_stmt
        | break_stmt
        | continue_stmt
        | expression_stmt ;

    let_stmt
        = "using"? "let" IDENTIFIER optional_type initializer? ";" ;

    const_stmt
        = "const" IDENTIFIER optional_type initializer? ";" ;

    return_stmt
        = "return" expression? ";" ;

    if_stmt
        = "if" "(" expression ")" block else_clause? ;

    else_clause
        = "else" if_stmt
        | "else" block ;

    while_stmt
        = "while" "(" expression ")" block ;

    for_stmt
        = "for" "(" for_initializer? ";" expression? ";" expression? ")" block ;

    for_initializer
        = let_stmt-without-semicolon
        | const_stmt-without-semicolon
        | expression ;

    foreach_stmt
        = "foreach" foreach_binding_list "in" expression block ;

    foreach_binding_list
        = foreach_value_binding
        | foreach_index_binding "," foreach_value_binding
        | foreach_key_binding "=>" foreach_value_binding
        | foreach_index_binding "," foreach_key_binding "=>" foreach_value_binding ;

    foreach_index_binding
        = IDENTIFIER optional_type ;

    foreach_key_binding
        = IDENTIFIER optional_type ;

    foreach_value_binding
        = "const"? "&"? IDENTIFIER optional_type ;

    switch_stmt
        = "switch" "(" expression ")" "{" switch_case* default_case? "}" ;

    switch_case
        = "case" expression ":" statement* ;

    default_case
        = "default" ":" statement* ;

    match_stmt
        = "match" expression "{" match_arm* "}" ;

    match_arm
        = IDENTIFIER ":" IDENTIFIER "=>" statement-or-block
        | IDENTIFIER "=>" statement-or-block ;

    statement-or-block
        = block
        | statement ;

    break_stmt
        = "break" ";" ;

    continue_stmt
        = "continue" ";" ;

    expression_stmt
        = expression ";" ;


Expressions
-----------

The parser reads balanced token slices and parses expressions through
ExpressionTokenParser. If token parsing fails, the parser reports a diagnostic
and leaves a RawExpressionNode; there is no secondary string-expression parser.

    expression
        = assignment ;

    assignment
        = conditional (assignment_operator assignment)? ;

    assignment_operator
        = "=" | "+=" | "-=" | "*=" | "/=" | "%=" ;

    conditional
        = range ("?" expression ":" conditional)? ;

    range
        = logical_or ((".." | "...") logical_or)? ;

    logical_or
        = logical_and ("||" logical_and)* ;

    logical_and
        = bitwise_or ("&&" bitwise_or)* ;

    bitwise_or
        = bitwise_xor ("|" bitwise_xor)* ;

    bitwise_xor
        = bitwise_and ("^" bitwise_and)* ;

    bitwise_and
        = equality ("&" equality)* ;

    equality
        = comparison (("==" | "!=") comparison)* ;

    comparison
        = shift (("<" | "<=" | ">" | ">=" | "<=>") shift)* ;

    shift
        = additive (("<<" | ">>") additive)* ;

    additive
        = multiplicative (("+" | "-") multiplicative)* ;

    multiplicative
        = unary (("*" | "/" | "%") unary)* ;

    unary
        = ("!" | "~" | "-" | "+" | "*" | "&" | "++" | "--") unary
        | postfix ;

    postfix
        = primary postfix_part* ;

    postfix_part
        = "(" argument_list? ")"
        | generic_arguments "(" argument_list? ")"
        | "." IDENTIFIER
        | "[" expression "]"
        | "++"
        | "--" ;

    primary
        = literal
        | IDENTIFIER
        | "true"
        | "false"
        | "null"
        | "sizeof" "(" (type | expression) ")"
        | "(" type ")" unary
        | "(" expression ")"
        | initializer_expression
        | function_expression ;

    literal
        = NUMBER
        | STRING
        | CHARACTER ;

    argument_list
        = expression ("," expression)* ;

    initializer_expression
        = type? "{" initializer_item_list? "}" ;

    initializer_item_list
        = initializer_item ("," initializer_item)* ","? ;

    initializer_item
        = IDENTIFIER ":" expression
        | expression ;

    function_expression
        = "fn" "(" parameter_list? ")" return_type? "=>" expression
        | "fn" "(" parameter_list? ")" return_type? block ;


Known parser cleanup targets
----------------------------

    1. Parse attribute argument values as expression nodes or richer token slices.
    2. Keep TypeNode as syntax, but avoid reparsing TypeNode.TypeName when TypeSyntax is available.
    3. Continue moving C/lowering string helpers toward typed AST or C AST nodes.
